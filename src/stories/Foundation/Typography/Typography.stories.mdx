export const LatoTextBlock = ({
  type,
  weight
}) => (
  <div className={['lato-text', `lato-text--${type}`, `lato-text--${type}_${weight}`].join(' ')}>
    Lato {type ? type.replaceAll('-', ' ') : ''}
  </div>
)
